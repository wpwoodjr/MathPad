// Degenerate vector triangle: shows Sweep 1 FAILED on one equation,

--Equations--

// Law of sines (solve for one magnitude given the other)
speed = smg*sin(cmg - set) / sin(cts - set)
drift = smg*sin(cts - cmg) / sin(cts - set)

// Solve for smg given cmg (quadratic form)
smg = drift*cos(cmg - set) + sqrt(speed**2 - drift**2*sin(cmg - set)**2)

// Solve for smg and cmg from cts (vector addition)
smg = sqrt((speed*sin(cts) + drift*sin(set))**2 + (speed*cos(cts) + drift*cos(set))**2)
cmg °= atan(speed*sin(cts) + drift*sin(set); speed*cos(cts) + drift*cos(set))

// Solve for speed and cts from smg/cmg (vector subtraction of current)
speed = sqrt((smg*sin(cmg) - drift*sin(set))**2 + (smg*cos(cmg) - drift*cos(set))**2)
cts °= atan(smg*sin(cmg) - drift*sin(set); smg*cos(cmg) - drift*cos(set))

// Solve for drift and set from smg/cmg (vector subtraction of boat)
drift = sqrt((smg*sin(cmg) - speed*sin(cts))**2 + (smg*cos(cmg) - speed*cos(cts))**2)
set °= atan(smg*sin(cmg) - speed*sin(cts); smg*cos(cmg) - speed*cos(cts))
