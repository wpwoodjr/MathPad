// Without fractional output, seconds round to nearest. The carries should
// propagate cleanly: 59.5s → +1 min, 59:59.5 → +1 hr, 23:59:59.5 → +1 day-ish.
// Fractional output (->>) keeps flooring so the .x part is shown verbatim.

a@t: 0:00:46.623
a@t->                "rounds up: 0:00:47"
a@t->>               "fractional: 0:00:46.623"

b@t: 0:00:59.5

d@t: 1:59:59.5
d@t->                "carries to 2:00:00"

e@t: 0:00:46.4
e@t->                "rounds down: 0:00:46"
