// Leeway correction
ws = speed*cos(leeway)

// Solve smg given drift, set (quadratic form)
smg = drift*cos(cmg - set) + sqrt(ws**2 - drift**2*sin(cmg - set)**2)

// Solve cts given smg, drift, set (vector subtraction of current)
cts = mod(atan(smg*sin(cmg) - drift*sin(set); smg*cos(cmg) - drift*cos(set)); 360)

// Solve drift and set given smg, cts (vector subtraction of boat)
drift = sqrt((smg*sin(cmg) - ws*sin(cts))**2 + (smg*cos(cmg) - ws*cos(cts))**2)
set = mod(atan(smg*sin(cmg) - ws*sin(cts); smg*cos(cmg) - ws*cos(cts)); 360)

// Law of sines (solve for drift given smg, cts, set)
drift = smg*sin(cts - cmg) / sin(cts - set)

// Heading from water track
hdg = cts - leeway
