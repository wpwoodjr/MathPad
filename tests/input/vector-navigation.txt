"1. Find current (drift, set)"
// Solved - same-vector pair

// Ground track = Boat velocity + Current
// Provide any 2 complete vectors, solve for the 3rd
// Also solves cross-vector pairs

// Course to steer: quadratic for smg (must be first - evaluates when cmg known but cts unknown)
smg = drift*cos(cmg - set) + sqrt(speed**2 - drift**2*sin(cmg - set)**2)

smg = sqrt((speed*sin(cts) + drift*sin(set))**2 + (speed*cos(cts) + drift*cos(set))**2)
cmg = mod(atan(speed*sin(cts) + drift*sin(set); speed*cos(cts) + drift*cos(set)); 360)

speed = sqrt((smg*sin(cmg) - drift*sin(set))**2 + (smg*cos(cmg) - drift*cos(set))**2)
cts = mod(atan(smg*sin(cmg) - drift*sin(set); smg*cos(cmg) - drift*cos(set)); 360)

drift = sqrt((smg*sin(cmg) - speed*sin(cts))**2 + (smg*cos(cmg) - speed*cos(cts))**2)
set = mod(atan(smg*sin(cmg) - speed*sin(cts); smg*cos(cmg) - speed*cos(cts)); 360)

// Law of sines: for when all 3 directions known, find magnitudes
speed = smg*sin(cmg - set) / sin(cts - set)
drift = smg*sin(cts - cmg) / sin(cts - set)
